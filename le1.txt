-- Load UI Library
local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/xHeptc/Kavo-UI-Library/main/source.lua"))()
local Window = Library.CreateLib("GALAXIUM CONTROL CORE", "BloodTheme")

-- Tab System
local tabAutoFarm = Window:NewTab("AUTO FARM")
local tabPVP = Window:NewTab("PVP")
local tabFruits = Window:NewTab("FRUITS")
local tabTeleport = Window:NewTab("TELEPORT")
local tabSeaEvent = Window:NewTab("SEA EVENT")
local tabMisc = Window:NewTab("MISC")
local tabESP = Window:NewTab("ESP")

-- AUTO FARM
local secFarm = tabAutoFarm:NewSection("Auto Farm Options")
secFarm:NewToggle("Auto Farm Level", "Tự đánh quái theo level", function(state)
    if state then
        print("Auto Farm Level: ON")
        -- Gọi hàm farm tại đây
    else
        print("Auto Farm Level: OFF")
    end
end)

secFarm:NewToggle("Auto Farm Boss", "Tự tìm boss, đánh và né", function(state)
    print("Auto Boss Farm: " .. tostring(state))
end)

-- PVP
local secPVP = tabPVP:NewSection("Combat Features")
secPVP:NewToggle("Aimbot Skill", "Tự canh tọa độ skill", function(state)
    print("Aimbot Skill: " .. tostring(state))
end)

secPVP:NewToggle("Kill Aura", "Tấn công đối thủ gần", function(state)
    print("Kill Aura: " .. tostring(state))
end)

-- FRUITS
local secFruit = tabFruits:NewSection("Fruit Features")
secFruit:NewButton("Auto Buy Random Fruit", "Mua trái ngẫu nhiên liên tục", function()
    print("Buying random fruit...")
end)

secFruit:NewToggle("Auto Store Fruit", "Nhặt và cất trái rơi", function(state)
    print("Auto Store: " .. tostring(state))
end)

-- TELEPORT
local secTP = tabTeleport:NewSection("Teleport Options")
secTP:NewDropdown("Teleport NPC", "Chọn NPC", {"Blox Dealer", "Sword Shop", "Colosseum"}, function(npc)
    print("Teleporting to: " .. npc)
end)

secTP:NewButton("Fly Hack (ON)", "Bay tự do", function()
    print("Fly Hack activated")
end)

-- SEA EVENT
local secSea = tabSeaEvent:NewSection("Sea Event Control")
secSea:NewToggle("Auto Sea Beast Farm", "Tự đánh Sea Beast", function(state)
    print("Sea Beast Farm: " .. tostring(state))
end)

secSea:NewButton("Teleport to Sea Event", "Dịch chuyển tới sự kiện", function()
    print("Teleporting to event...")
end)

-- ESP
local secESP = tabESP:NewSection("ESP Settings")
secESP:NewToggle("ESP Players", "Hiện thông tin người chơi", function(state)
    print("ESP Players: " .. tostring(state))
end)

secESP:NewToggle("ESP Chests & Fruits", "Hiện trái & rương", function(state)
    print("ESP Chests: " .. tostring(state))
end)

-- MISC
local secMisc = tabMisc:NewSection("Misc & Settings")

secMisc:NewButton("God Mode", "Bất tử (tuỳ executor)", function()
    print("God Mode enabled")
end)

-- UI TOGGLE + Custom Keybind
secMisc:NewKeybind("Toggle UI", "Ẩn/hiện UI", Enum.KeyCode.LeftControl, function()
    Library:ToggleUI()
end)

secMisc:NewTextBox("Thay phím UI", "Nhập tên phím (vd: F)", function(input)
    local success, newKey = pcall(function()
        return Enum.KeyCode[input]
    end)
    if success and newKey then
        secMisc:NewKeybind("UI Toggle (" .. input .. ")", "Phím mới để bật/tắt UI", newKey, function()
            Library:ToggleUI()
        end)
        print("Đã thêm phím bật/tắt mới: " .. input)
    else
        warn("Phím không hợp lệ!")
    end
end)
